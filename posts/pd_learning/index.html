pandas之数据分析

先来看看基本的使用,先读取一个表格数据

import numpy as np
import pandas as pd

# 导入csv文件或者excel文件
df = pd.DataFrame(pd.read_csv('./data/test.csv',header=1))
# df = pd.DataFrame(pd.read_excel('name.xlsx'))

当然，表格数据可以通过文件读写，将其打开，并将行和列整理好叫给pd处理

import pandas as pd
from collections import namedtuple

Item = namedtuple('Item', ['reply', 'pv'])
items = []

with open('reply.pv.07', 'r', encoding='utf-8') as f:
    for line in f:
        line_split = line.strip().split('\t')
        items.append(Item(line_split[0].strip(), line_split[1].strip()))

df = pd.DataFrame(items, columns=['reply', 'pv'])=

我们还可以通过pd，来创建一个数据表

import numpy as np
